44 bool JointReader::Init(std::string part, double sigma, unsigned int pop_size, double deg_per_neuron, std::string ini_path) {
45  /*
46  Initialize the joint reader with given parameters
47 
48  params: std::string part -- string representing the robot part, has to match iCub part naming {left_(arm/leg), right_(arm/leg), head, torso}
49  sigma -- sigma for the joints angles populations coding
50  int pop_size -- number of neurons per population, encoding each one joint angle; only works if parameter "deg_per_neuron" is not set
51  double deg_per_neuron -- degree per neuron in the populations, encoding the joints angles; if set: population size depends on joint working range
52 
53  return: bool -- return True, if successful
54  */
55 
56  if (!this->dev_init) {
57  if (!CheckPartKey(part)) {
58  std::cerr << "[Joint Reader] " << part << " is an invalid iCub part key!" << std::endl;
59  return false;
60  }
61  this->icub_part = part;
62 
63  if (pop_size < 0) {
64  std::cerr << "[Joint Reader " << icub_part << "] Population size have to be positive!" << std::endl;
65  return false;
66  }
67  if (sigma < 0) {
68  std::cerr << "[Joint Reader " << icub_part << "] Sigma have to be positive!" << std::endl;
69  return false;
70  }
71  sigma_pop = sigma;
72 
73  if (!yarp::os::Network::checkNetwork()) {
74  std::cerr << "[Joint Reader " << icub_part << "] YARP Network is not online. Check nameserver is running!" << std::endl;
75  return false;
76  }
77 
78  // read configuration data from ini file
79  INIReader reader_gen(ini_path + "interface_param.ini");
80  bool on_Simulator = reader_gen.GetBoolean("general", "simulator", true);
81  std::string robot_port_prefix = reader_gen.Get("general", "robot_port_prefix", "/icubSim");
82  if (on_Simulator && (robot_port_prefix != "/icubSim")) {
83  std::cerr << "[Joint Reader " << icub_part << "] The port prefix does not match the default simulator prefix!" << std::endl;
84  return false;
85  }
86  std::string client_port_prefix = reader_gen.Get("general", "client_port_prefix", "/client");
87 
88  // setup iCub joint position control
89  yarp::os::Property options;
90  options.put("device", "remote_controlboard");
91  options.put("remote", (robot_port_prefix + "/" + icub_part).c_str());
92  options.put("local", (client_port_prefix + "/ANNarchy_read/" + icub_part).c_str());
93 
94  if (!driver.open(options)) {
95  std::cerr << "[Joint Reader " << icub_part << "] Unable to open " << options.find("device").asString() << "!" << std::endl;
96  return false;
97  }
98 
99  if (!driver.view(ienc)) {
100  std::cerr << "[Joint Reader " << icub_part << "] Unable to open motor control interfaces!" << std::endl;
101  return false;
102  }
103 
104  ienc->getAxes(&joints);
105  joint_angles.resize(joints);
106  neuron_deg.resize(joints);
107  joint_deg_res.resize(joints);
108 
109  // setup ini-Reader for joint limits
110  double joint_range;
111  std::string joint_path;
112  if (on_Simulator) {
113  joint_path = reader_gen.Get("motor", "sim_joint_limits_path", "");
114  } else {
115  joint_path = reader_gen.Get("motor", "robot_joint_limits_path", "");
116  }
117 
118  if (joint_path == "") {
119  std::cerr << "[Joint Reader " << icub_part << "] Ini-path for joint limits is empty!" << std::endl;
120  return false;
121  }
122  INIReader reader(joint_path);
123 
124  for (int i = 0; i < joints; i++) {
125  // read joint limits
126  joint_min.push_back(reader.GetReal(icub_part.c_str(), ("joint_" + std::to_string(i) + "_min").c_str(), 0.0));
127  joint_max.push_back(reader.GetReal(icub_part.c_str(), ("joint_" + std::to_string(i) + "_max").c_str(), 0.0));
128 
129  // printf("joint: %i, min: %f, max: %f \n", i, joint_min[i], joint_max[i]);
130 
131  if (joint_min == joint_max) {
132  std::cerr << "[Joint Reader " << icub_part << "] Error in reading joint parameters from .ini file!" << std::endl;
133  return false;
134  }
135 
136  // compute population code resolution
137  joint_range = joint_max[i] - joint_min[i] + 1;
138 
139  if (pop_size > 0) { // with given population size
140  neuron_deg[i].resize(pop_size);
141  joint_deg_res[i] = joint_range / pop_size;
142 
143  for (unsigned int j = 0; j < neuron_deg[i].size(); j++) {
144  neuron_deg[i][j] = joint_min[i] + j * joint_deg_res[i];
145  }
146 
147  } else if (deg_per_neuron > 0.0) { // with given neuron resolution
148  int joint_res = std::floor(joint_range / deg_per_neuron);
149  neuron_deg[i].resize(joint_res);
150  joint_deg_res[i] = deg_per_neuron;
151 
152  for (unsigned int j = 0; j < neuron_deg[i].size(); j++) {
153  neuron_deg[i][j] = joint_min[i] + j * deg_per_neuron;
154  }
155  } else {
156  std::cerr << "[Joint Reader " << icub_part
157  << "] Error in population size definition. Check the values for pop_size or deg_per_neuron!" << std::endl;
158  return false;
159  }
160  }
161 
162  this->type = "JointReader";
163  this->dev_init = true;
164  return true;
165  } else {
166  std::cerr << "[Joint Reader " << icub_part << "] Initialization aready done!" << std::endl;
167  return false;
168  }
169 }

539 std::vector<double> JointReader::Encode(double joint_angle, int joint) {
540  /*
541  Encode given joint position into a vector
542

545

558 double JointReader::NormalPdf(double value, double mean, double sigma) {
559  /*
560  Return the normal distribution value for a given value, mean and sigma
561 
562  params: double value -- value to calculate normal distribution at
563  double mean -- mean of the normal distribution
564  double sigma -- sigma of the normal distribution
565 
566  return: double -- function value for the normal distribution
567  */
568 
569  // double inv_sqrt_2pi = 4.0 / (sigma * std::sqrt(2 * M_PI));
570  double a = (value - mean) / sigma;
571 
572  // return inv_sqrt_2pi * std::exp(-0.5 * a * a);
573  return 1.0 * std::exp(-0.5 * a * a);
574 }
