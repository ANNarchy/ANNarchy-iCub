205 void SkinReader::Close() {
206  /*
207  Close and clean skin reader
208  */
209 
210  if (!port_hand.isClosed()) {
211  yarp::os::Network::disconnect(("/icubSim/skin/" + side + "_hand_comp").c_str(), ("/Skin_Reader/" + side + "_hand:i").c_str());
212  port_hand.close();
213  }
214  if (!port_forearm.isClosed()) {
215  yarp::os::Network::disconnect(("/icubSim/skin/" + side + "_forearm_comp").c_str(), ("/Skin_Reader/" + side + "_forearm:i").c_str());
216  port_forearm.close();
217  }
218  if (!port_arm.isClosed()) {
219  yarp::os::Network::disconnect(("/icubSim/skin/" + side + "_arm_comp").c_str(), ("/Skin_Reader/" + side + "_arm:i").c_str());
220  port_arm.close();
221  }
222 #ifdef _USE_GRPC
223  if (dev_init_grpc) {
224  skin_source->shutdown();
225  server_thread.join();
226  delete skin_source;
227  dev_init_grpc = false;
228  }
229 #endif
230  this->dev_init = false;
231 }

302  if (CheckInit()) {

306 
307  bool no_error = true;
308 
309  // HAND

413 bool SkinReader::ReadTaxelPos(std::string filename_idx, std::string filename_pos, std::string part) {
414  /*
415  Read information about taxel position from config files
416 
417  params: std::string filename_idx -- filename, containing the path, of the file with the taxel index data
418  std::string filename_pos -- filename, containing the path, of the file with the taxel position data
419  std::string part -- skin part to load the data for (e.g. forearm)
420 
421  return: bool -- return True, if successful
422  */
423 
424  std::ifstream file_idx;
425  std::ifstream file_pos;
426  std::vector<int> idx;
427  std::vector<std::vector<double>> arr;
428  std::vector<double> pos;
429  TaxelData part_tax_pos;
430  int ind;
431  double val;
432  int i;
433 
434  // read information from file about used indices of the skin sensor position data
435  i = 0;
436  file_idx.open(filename_idx);
437  if (file_idx.fail()) {
438  std::cerr << "[Skin Reader " + side + "] Could not open taxel index file!" << std::endl;
439  return false;
440  }
441  while (file_idx >> ind) {
442  i++;
443  idx.push_back(ind);
444  }
445  file_idx.close();
446 
447  // read information about skin sensor (also called taxel) position
448  i = 0;
449  file_pos.open(filename_pos);
450  if (file_pos.fail()) {
451  std::cerr << "[Skin Reader " + side + "] Could not open taxel position file!" << std::endl;
452  return false;
453  }
454  while (file_pos >> val) {
455  i++;
456  i = i % 6;
457  if (i < 3) {
458  pos.push_back(val);
459  }
460  if (i % 6 == 0) {
461  arr.push_back(pos);
462  pos.clear();
463  }
464  }
465  file_pos.close();
466 
467  part_tax_pos.idx = idx;
468  part_tax_pos.arr = arr;
469 
470  taxel_pos_data[part] = part_tax_pos;
471 
472  return true;
473 }
