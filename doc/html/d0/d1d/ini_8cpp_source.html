52 /* Return pointer to first char (of chars) or inline comment in given string,
53  or pointer to null at end of string if neither found. Inline comment must
54  be prefixed by a whitespace character to register as a comment. */
55 static char* find_chars_or_comment(const char* s, const char* chars)
56 {
57 #if INI_ALLOW_INLINE_COMMENTS
58  int was_space = 0;
59  while (*s && (!chars || !strchr(chars, *s)) &&
60  !(was_space && strchr(INI_INLINE_COMMENT_PREFIXES, *s))) {
61  was_space = isspace((unsigned char)(*s));
62  s++;
63  }
64 #else
65  while (*s && (!chars || !strchr(chars, *s))) {
66  s++;
67  }
68 #endif
69  return (char*)s;
70 }

81 int ini_parse_stream(ini_reader reader, void* stream, ini_handler handler,
82  void* user)
83 {
84  /* Uses a fair bit of stack (use heap instead if you need to) */
85 #if INI_USE_STACK
86  char line[INI_MAX_LINE];
87  int max_line = INI_MAX_LINE;
88 #else
89  char* line;
90  int max_line = INI_INITIAL_ALLOC;
91 #endif
92 #if INI_ALLOW_REALLOC
93  char* new_line;
94  int offset;
95 #endif
96  char section[MAX_SECTION] = "";
97  char prev_name[MAX_NAME] = "";
98 
99  char* start;
100  char* end;
101  char* name;
102  char* value;
103  int lineno = 0;
104  int error = 0;
105 
106 #if !INI_USE_STACK
107  line = (char*)malloc(INI_INITIAL_ALLOC);
108  if (!line) {
109  return -2;
110  }
111 #endif
112 
113 #if INI_HANDLER_LINENO
114 #define HANDLER(u, s, n, v) handler(u, s, n, v, lineno)
115 #else
116 #define HANDLER(u, s, n, v) handler(u, s, n, v)
117 #endif
118 
119  /* Scan through stream line by line */
120  while (reader(line, max_line, stream) != NULL) {
121 #if INI_ALLOW_REALLOC
122  offset = strlen(line);
123  while (offset == max_line - 1 && line[offset - 1] != '\n') {
124  max_line *= 2;
125  if (max_line > INI_MAX_LINE)
126  max_line = INI_MAX_LINE;
127  new_line = realloc(line, max_line);
128  if (!new_line) {
129  free(line);
130  return -2;
131  }
132  line = new_line;
133  if (reader(line + offset, max_line - offset, stream) == NULL)
134  break;
135  if (max_line >= INI_MAX_LINE)
136  break;
137  offset += strlen(line + offset);
138  }
139 #endif
140 
141  lineno++;
142 
143  start = line;
144 #if INI_ALLOW_BOM
145  if (lineno == 1 && (unsigned char)start[0] == 0xEF &&
146  (unsigned char)start[1] == 0xBB &&
147  (unsigned char)start[2] == 0xBF) {
148  start += 3;
149  }
150 #endif
151  start = lskip(rstrip(start));
152 
153  if (strchr(INI_START_COMMENT_PREFIXES, *start)) {
154  /* Start-of-line comment */
155  }
156 #if INI_ALLOW_MULTILINE
157  else if (*prev_name && *start && start > line) {
158  /* Non-blank line with leading whitespace, treat as continuation
159  of previous name's value (as per Python configparser). */
160  if (!HANDLER(user, section, prev_name, start) && !error)
161  error = lineno;
162  }
163 #endif
164  else if (*start == '[') {
165  /* A "[section]" line */
166  end = find_chars_or_comment(start + 1, "]");
167  if (*end == ']') {
168  *end = '\0';
169  strncpy0(section, start + 1, sizeof(section));
170  *prev_name = '\0';
171  }
172  else if (!error) {
173  /* No ']' found on section line */
174  error = lineno;
175  }
176  }
177  else if (*start) {
178  /* Not a comment, must be a name[=:]value pair */
179  end = find_chars_or_comment(start, "=:");
180  if (*end == '=' || *end == ':') {
181  *end = '\0';
182  name = rstrip(start);
183  value = end + 1;
184 #if INI_ALLOW_INLINE_COMMENTS
185  end = find_chars_or_comment(value, NULL);
186  if (*end)
187  *end = '\0';
188 #endif
189  value = lskip(value);
190  rstrip(value);
191 
192  /* Valid name[=:]value pair found, call handler */
193  strncpy0(prev_name, name, sizeof(prev_name));
194  if (!HANDLER(user, section, name, value) && !error)
195  error = lineno;
196  }
197  else if (!error) {
198  /* No '=' or ':' found on name[=:]value line */
199  error = lineno;
200  }
201  }
202 
203 #if INI_STOP_ON_FIRST_ERROR
204  if (error)
205  break;
206 #endif
207  }
208 
209 #if !INI_USE_STACK
210  free(line);
211 #endif
212 
213  return error;
214 }

236 /* An ini_reader function to read the next line from a string buffer. This
237  is the fgets() equivalent used by ini_parse_string(). */
238 static char* ini_reader_string(char* str, int num, void* stream) {
239  ini_parse_string_ctx* ctx = (ini_parse_string_ctx*)stream;
240  const char* ctx_ptr = ctx->ptr;
241  size_t ctx_num_left = ctx->num_left;
242  char* strp = str;
243  char c;
244 
245  if (ctx_num_left == 0 || num < 2)
246  return NULL;
247 
248  while (num > 1 && ctx_num_left != 0) {
249  c = *ctx_ptr++;
250  ctx_num_left--;
251  *strp++ = c;
252  if (c == '\n')
253  break;
254  num--;
255  }
256 
257  *strp = '\0';
258  ctx->ptr = ctx_ptr;
259  ctx->num_left = ctx_num_left;
260  return str;
261 }
